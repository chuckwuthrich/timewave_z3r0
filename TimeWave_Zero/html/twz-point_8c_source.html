176 /*--------------*/
177 long double f(long double x,
178  int64_t number_set)
179 {
180 register i;
181 long double sum = 0.0, last_sum = 0.0;
182 
183 if ( x )
184  {
185  for ( i=0; x>=powers[i]; i++ )
186  sum += mult_power(v(div_power(x,i),number_set),i);
187 
188  i = 0;
189  do
190  {
191  if ( ++i > CALC_PREC+2 )
192  break;
193  last_sum = sum;
194  sum += div_power(v(mult_power(x,i),number_set),i);
195  } while ( ( sum == 0.0 ) || ( sum > last_sum ) );
196  }
197 
198 /* dividing by 64^3 gives values consistent with the Apple // version
199  * and provides more convenient y-axis labels
200  */
201 sum = div_power(sum,3);
202 
203 return ( sum );
204 }

207 long double v(long double y,
208  int64_t number_set)
209 {
210 int64_t i = (int64_t)(fmod(y,(long double)NUM_DATA_POINTS));
211 int64_t j = (i+1)%NUM_DATA_POINTS;
212 long double z = y - floor(y);
213 
214 return ( z==0.0 ? (long double)w[number_set][i] :
215  ( w[number_set][j] - w[number_set][i] )*z + w[number_set][i] );
216 }

218 /* in order to speed up the calculation, if wave factor = 64
219  * then instead of using multiplication or division operation
220  * we act directly on the floating point representation;
221  * multiplying by 64^i is accomplished by adding i*0x60
222  * to the exponent (the last 2 bytes of the 8-byte representation);
223  * dividing by 64^i is accomplished by subtracting i*0x60
224  * from the exponent
225  */

228 long double mult_power(long double x,
229  int64_t i)
230 {
231 /* Removing this code: Swithing to 64-bit datatypes

233 if ( wave_factor == 64 )

246 /* Removing trick: moving to 64-bit datatypes

249  *exponent -= i*0x60;

251 */
