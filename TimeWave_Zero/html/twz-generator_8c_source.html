327 long double
328 v (long double y,
329  int64_t number_set)
330 {
331 
332  int64_t i = (int64_t) (fmod (y, (long double) NUM_DATA_POINTS));
333 
334  int64_t j = (i + 1) % NUM_DATA_POINTS;
335 
336  long double z = y - floor (y);
337 
338 
339  return (z == 0.0 ? (long double) w[number_set][i] :
340  (w[number_set][j] - w[number_set][i]) * z + w[number_set][i]);
341 
342 }

345 /* in order to speed up the calculation, if wave factor = 64
346 * then instead of using multiplication or division operation
347 * we act directly on the floating point representation;
348 * multiplying by 64^i is accomplished by adding i*0x60
349 * to the exponent (the last 2 bytes of the 8-byte representation);
350 * dividing by 64^i is accomplished by subtracting i*0x60
351 * from the exponent
352 */
